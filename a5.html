        // Including the Capacitive Sensor library into my code
        #include 
        // sets the capacitive sensor between pins 4 and 2. 2 is the sensor.
        CapacitiveSensor capSensor = CapacitiveSensor(4,2);
        // Setting the gate pin for the transistor
        const int gate = 9;
      
        // Sets up the code
        void setup() {
          // Initializes the LED pin as an output pin. The pin currrently has no voltage supply and it would be written to it as we progress with the code.
          pinMode(gate, OUTPUT);
          // initializes tbhe capacitive sensor
          capSensor.set_CS_AutocaL_Millis(0xFFFFFFFF);
      
        }
      
        // executes this code again and again until the Arduino has power
        void loop() {
          // Reads the touch sensor value
          long touchValue = capSensor.capacitiveSensor(30);
          // maps the touch value to an analog value for the gate pin.
          int output = map(touchValue, 0, 1023, 0, 255);
          // sends the output value to the transistor through the gate pin.
          analogWrite(gate, output);
        }


Circuit's Operation



The video shows the working of the circuit. The value detected from the capacitive sensor determines the voltage through the gate pin of the transistor. This determines how much current would flow through the circuit.
